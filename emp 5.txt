package MultipleInterface;

class MultipleInterfaceDemo{ 
    public static void main(String[] args) { 
 
        Employee e1 = new Employee("John", 21, 'M',"Google","Manager", 1); 
 
        e1.p_display(); 
        e1.o_display(); 
    } 
} 
 
interface Personal { 
    void p_display(); 
} 
 
interface Official { 
    void o_display(); 
} 
 
class Employee implements Personal, Official { 
 
    String name; 
    int age; 
    char gender; 
    String company; 
    String role; 
    int exp; 
 
    Employee(String name, int age, char gender, String company, String role, int exp) { 
 
        this.name = name; 
        this.age = age; 
        this.gender = gender; 
        this.company = company; 
        this.role = role; 
        this.exp = exp; 
    } 
 
    public void p_display() { 
        System.out.println("PERSONAL DETAILS"); 
        System.out.println("Name: " + name); 
        System.out.println("Age: " + age); 
        System.out.println("Gender: " + gender); 
    } 
 
    public void o_display() { 
        System.out.println("OFFICIAL DETAILS"); 
        System.out.println("Company: " + company); 
        System.out.println("Role: " + role); 
        System.out.println("Experience: " + exp); 
    } 
}
